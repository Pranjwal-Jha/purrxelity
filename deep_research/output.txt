import { Textarea } from "@/components/ui/textarea";
import { Button } from "@/components/ui/button";
import { ArrowUp, Lightbulb, Paperclip } from "lucide-react";

interface ChatInputProps {
  input: string;
  setInput: (value: string) => void;
  handleSendMessage: () => void;
  isDeepThink: boolean;
  onToggleDeepThink: () => void;
}

export function ChatInput({
  input,
  setInput,
  handleSendMessage,
  isDeepThink,
  onToggleDeepThink,
}: ChatInputProps) {
  const handleKeyDown = (e: React.KeyboardEvent<HTMLTextAreaElement>) => {
    if ((e.metaKey || e.ctrlKey) && e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSendMessage();
    }
  };

  return (
    <div className="p-4 border-t flex justify-center">
      <div className="w-1/2 relative">
        <div className="relative flex items-end rounded-lg border bg-background focus-within:ring-1 focus-within:ring-ring">
          <Textarea
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Type your message... (Cmd+Enter to send)"
            className="h-28 resize-none border-none bg-transparent shadow-none focus-visible:ring-0"
          />
          <div className="absolute bottom-2 right-2 flex items-center gap-1">
            <Button
              variant="ghost"
              size="icon"
              disabled // Placeholder button
            >
              <Paperclip className="h-5 w-5" />
            </Button>
            <Button
              variant={isDeepThink ? "secondary" : "ghost"}
              size="icon"
              onClick={onToggleDeepThink}
            >
              <Lightbulb className="h-5 w-5" />
            </Button>
            <Button
              size="icon"
              onClick={handleSendMessage}
              disabled={!input.trim()}
            >
              <ArrowUp className="h-5 w-5" />
            </Button>
          </div>
        </div>
      </div>
    </div>
  );
}
